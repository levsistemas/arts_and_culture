function actualizarUsuario(argumentos){
    fetch(`localhost:3000/edit/proyecto/editar/${id}/${id}`, {
        method: "PUT",
        headers: {
            "Content-Type": "application/json"
        },
        body: JSON.stringify({nombre: argumentos.nombre, apellido: argumentos.apellido, dni: argumentos.dni, celular: argumentos.celular, sexo: argumentos.sexo, email: argumentos.email})
    })
    .then(function (response){
        if(response.ok){
            console.log('Usuario actualizado correctamente...')
        } else {
            console.log('Error al actualizar usuario...')
        }
    })
    .catch(function (error) {
        console.log('Error en la solicitud: ' + error.message);
    })
}
